Форматирование чисел

/**
* Расставляет пробелы между разрядами и заменяет точку на запятую
*
* @param num {number | string} Исходное число
* @returns {void | string} Строка с отформатированным числом
*/
function formatNumber(num) {
    return num.replace(/(\d)(?=(\d\d\d)+([^\d]|$))/g, '$1 ').replace('.', ',');
}

Форматирование российских телефонных номеров

const telInputs = document.querySelectorAll('input[type=tel]');
for (let i = 0; i < telInputs.length; i++) {
    telInputs[i].addEventListener('input', function(e) {
        let cleaned = e.target.value.replace(/\D/g, '');
        if (['7','8'].indexOf(cleaned[0]) >= 0) {
            cleaned = cleaned.substring(1);
        }
        let match = cleaned.match(/^(\d{3})(\d{3})(\d{2})(\d{2})$/);
        if (match) {
            e.target.value = ['+7 ', match[1], ' ', match[2], '-', match[3], '-', match[4]].join('');
        }
    });
}

 * Переводит rem в пиксели
 *
 * @param rem {number} Кол-во rem
 * @returns {number} Сколько пикселей содержится в исходных rem

function remToPx(rem) {
    return rem * parseFloat(getComputedStyle(document.documentElement).fontSize);
}

PHP

Форматирование чисел

/**
* Расставляет неразрывные пробелы между разрядами и округляет дробную часть
*
* @param int $number Исходное число
*
* @return void|string Строка с отформатированным числом
*/
function number_format_groups($number) {
    return number_format($number, 0, ',', ' ');
}

Телефонные ссылки

/**
* Создание URL-ссылки из номера телефона 
*
* @param string $tel Номер телефона, например '+7 (999) 999-99-99'
*
* @return string URL для атрибута href, например 'tel:%2B79999999999'
*/
function telencode($tel) {
    return 'tel:'.str_replace(['+', ' ', ' ', '(', ')', '-'], ['%2B'], $tel);
}
